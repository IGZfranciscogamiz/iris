 97 	function _Window_OnHashChange () {
 98 		var prev = _PrevHashUrl.split("/")
 99 		,	curr = document.location.hash.split("/")
100 		,	prevPath = ""
101 		,	currPath = ""
102 		,	pathWithoutParams
103 		,	hasRemainingChilds = false
104 		;
105 		
106 		// Hide screens and its childs that are not showed
107 		if ( prev.length > curr.length ) {
108 			
109 			for ( var f=0, F=prev.length; f<F; f++ ) {
110 				prevPath += prev[f] + "/";
111 				
112 				if ( curr[f] ) {
113 					currPath += curr[f] + "/";
114 				}
115 				
116 				if ( hasRemainingChilds || currPath != prevPath ) {
117 					hasRemainingChilds = true;
118 					
119 					pathWithoutParams = _RemoveURLParams(prevPath);
120 					_Screen[pathWithoutParams].__Sleep__();
121 					_Screen[pathWithoutParams].Hide();
122 				}
123 			}
124 		}
125 		
126 		// Show child screens
127 		prevPath = "";
128 		currPath = "";
129 		hasRemainingChilds = false;
130 		for ( var f=0, F=curr.length; f<F; f++ ) {
131 			currPath += curr[f] + "/";
132 			
133 			if ( prev[f] ) {
134 				prevPath += prev[f] + "/";
135 			}
136 			
137 			if ( hasRemainingChilds || currPath != prevPath ) {
138 				hasRemainingChilds = true;
139 				
140 				pathWithoutParams = _RemoveURLParams(currPath);
141 				_ShowScreen(pathWithoutParams, _NavGetParams(curr[f]) );
142 				
143 			}
144 		}
145 		
146 		_PrevHashUrl = _RemoveLastSlash(currPath);
147 	}

615 	function _TemplateParse (p_html, p_data) {
616 		var result = p_html
617 		,	matches
618 		,	formatLabel
619 		,	value
620 		,	regExp = /##([0-9A-Za-z_\.]+)(?:\|(date|currency)(?:\(([^\)]+)\))*)?##/g
621 		;
622 
623 		while ( matches = regExp.exec(p_html) ) {
624 			value = p_data[matches[1]];
625 
626 			formatLabel = matches[2];
627 			if ( formatLabel ) {
628 				switch (formatLabel) {
629 					case "date":
630 						value = _DateFormat(value, matches[3]);
631 						break;
632 					case "currency":
633 						value = _ParseCurrency(value);
634 						break;
635 					default:
636 						iris.W("Unknow template format label", formatLabel);
637 				}
638 			}
639 
640 			result = result.replace(matches[0], value);
641 		}
642 		
643 		return result;
644 	}

646 	function _ParseCurrency (p_value) {
647 		var settings = _Regional[_Locale]["currency"];
648 		
649 		var val = Number(p_value);
650 		var format = (val >= 0) ? settings["formatPos"] : settings["formatNeg"];
651 		
652 		var decimal = val % 1;
653 		var num = String( Math.abs(val-decimal) );
654 		
655 		decimal = String(Math.abs(decimal).toFixed(settings["precision"]))
656 		decimal = decimal.substr(2);
657 	
658 		for (var i = 0; i < Math.floor((num.length - (1 + i)) / 3); i++) {
659 			num = num.substring(0, num.length - (4 * i + 3)) + settings["thousand"] + num.substring(num.length - (4 * i + 3));
660 		}
661 		
662 		return format.replace("n", num + settings["decimal"] + decimal );
663 	}

666 		if ( !p_format ) {
667 			p_format = _Regional[_Locale]["dateFormat"];
668 		}
669 		
670 		if ( typeof p_date !== "object" ) {
671 			p_date = new Date(Number(p_date))
672 		}
