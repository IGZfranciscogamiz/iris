Destroy a UI child from a parent container

The list item creation is explained here

Example template:

<div>
    <button data-id="btncreateUI">Create a new UI</button>

    <br>UI Index:
    <input data-id='idx' type='number' value='0' />
    <button data-id='btn_destroy'>Destroy</button>

    <br>Container:
    <div data-id='container'></div>
</div>

All UIs are registered into uiList Array:

iris.screen(function(self) {
    var uiList = []
    ,   inputIdx
    ;
    
    self.create = function() {
        self.tmpl("screen/example_destroy.html");

        self.get("btncreateUI").click(createUI);
        self.get("btn_destroy").click(deleteUI);

        inputIdx = self.get("idx");
        
    }

    function createUI() {
        uiList.push(
            self.ui("container", iris.path.ui_example)
        );
    }

    function deleteUI() {
        var idx = inputIdx.val();
        self.DestroyUI(uiList[idx]);
        uiList.splice(idx, 1);
    }
});
