185 	function _Window_OnHashChange () {
186 		
187 		if ( !_WelcomeScreenCreated ) {
188 			iris.E("You must set the welcome screen using iris.screen.WelcomeScreen()");
189 			return false;
190 		}
191 		
192 		if ( _GotoCancelled ) {
193 			_GotoCancelled = false;
194 			return false;
195 		}
196 		
197 		var prev = _PrevHashUrl.split("/")
198 		,	curr = document.location.hash.split("/")
199 		,	prevPath = ""
200 		,	currPath = ""
201 		,	pathWithoutParams
202 		,	hasRemainingChilds = false
203 		;
204 		
205 		// Check if all screen.canSleep() are true
206 		if ( _PrevHashUrl != "" ) {
207 			for ( var f=0, F=prev.length; f<F; f++ ) {
208 				
209 				if ( prev[f] != "" ) {
210 					prevPath += prev[f] + "/";
211 					pathWithoutParams = _RemoveURLParams(prevPath);
212 					
213 					if (_Screen.hasOwnProperty(pathWithoutParams) && _Screen[pathWithoutParams].CanSleep() === false ) {
214 						_GotoCancelled = true;
215 						document.location.hash = _PrevHashUrl;
216 						return false;
217 					}
218 				}
219 			}			
220 		}
221 		prevPath = "";
222 		
223 		// Hide screens and its childs that are not showed
224 		if ( prev.length > curr.length ) {
225 			
226 			for ( var f=0, F=prev.length; f<F; f++ ) {
227 				prevPath += prev[f] + "/";
228 				
229 				if ( curr[f] ) {
230 					currPath += curr[f] + "/";
231 				}
232 				
233 				if ( hasRemainingChilds || currPath != prevPath ) {
234 					hasRemainingChilds = true;
235 					pathWithoutParams = _RemoveURLParams(prevPath);
236 
237 					_Screen[pathWithoutParams].__Sleep__();
238 					_Screen[pathWithoutParams].Hide();
239 				}
240 			}
241 		}
242 		
243 		// Show child screens
244 		prevPath = "";
245 		currPath = "";
246 		hasRemainingChilds = false;
247 		for ( var f=0, F=curr.length; f<F; f++ ) {
248 			currPath += curr[f] + "/";
249 			
250 			if ( prev[f] ) {
251 				prevPath += prev[f] + "/";
252 			}
253 			
254 			if ( hasRemainingChilds || currPath != prevPath ) {
255 				hasRemainingChilds = true;
256 				
257 				pathWithoutParams = _RemoveURLParams(currPath);
258 				_ShowScreen(pathWithoutParams, _NavGetParams(curr[f]) );
259 				
260 			}
261 		}
262 		
263 		_PrevHashUrl = _RemoveLastSlash(currPath);
264 	};

351 							script.language = "javascript";
352 							script.type = "text/javascript";
353 							script.text = p_data;
354 							_Head.appendChild(script);
355 						}
356 					}
357 					, function (p_err) {
358 						delete _Includes[fileUrl];
359 						_E(p_err.status, "Error loading file '" + fileUrl + "'");
360 					}
361 				);
